window.onscroll = function(){ var t=document.documentElement.scrollTop||document.body.scrollTop; var toTop_div=document.querySelector(".toTop"); if(t>=100){ toTop_div.style.display="block" }else{ toTop_div.style.display="none" } }
$(".toTop").click(function(){ $("body").animate({"scrollTop":"0"},200) })
